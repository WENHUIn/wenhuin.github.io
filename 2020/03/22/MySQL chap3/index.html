inner/outer join | 内外连接

# 同一个database的连接
select ..
from table1
join table2   
    on 条件
# or 
# join ... using () 必须有同名的key
# 不同database的连接
select ..
from database1.table
join database2.table 
    on 条件
# self joins 自连接
# 同一个table里，如要向谁报道
select 
   ...,
   ...,
   ...,
from table as a
left join table b
    on a.XX = b.XX 

# 多表连接
select ..
from table1
join table2 
    on 条件
join table3
    on 条件
# 多表外连接
# left join: 左表内容完全输出无论条件是否成立
# right join: 右表内容完全输出无论条件是否成立
select ..
from table1
left join table2 
    on 条件
left join table3
    on 条件



# 将多个Table直接组合在一起 不需要使用key连接

# 1. implicitly
select *
from shippers,products
# 2. explicitly
select *
from shippers
cross join products p 
# UNION | combine rows 
select ..
from ..
where ..
union
select ..
from ..
where ..
